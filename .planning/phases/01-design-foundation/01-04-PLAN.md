    **Step B: Create src/components/layout/Layout.tsx**

    A minimal Server Component that composes Header and Footer around children:

    ```tsx
    import { Header } from './Header';
    import { Footer } from './Footer';

    interface LayoutProps {
      children: React.ReactNode;
    }

    export function Layout({ children }: LayoutProps) {
      return (
        <>
          <Header />
          <main className="flex-1">
            {children}
          </main>
          <Footer />
        </>
      );
    }

    export default Layout;
    ```

    Note: `flex-1` on `<main>` works because the root `<body>` already has `flex flex-col` from layout.tsx. This pushes the footer to the bottom even on short pages.
